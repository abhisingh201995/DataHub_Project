package com.DataObjects.DhssDataObject;

public class ViewWrapperAndPipelineHistoryDTO {

		this.edapTenantId = edapTenantId;
	}

	public String getTenantId() {
		return tenantId;
	}

	public void setTenantId(String tenantId) {
		this.tenantId = tenantId;
	}

	public String getShortName() {
		return shortName;
	}

	public void setShortName(String shortName) {
		this.shortName = shortName;
	}

	public List<JobstatusDTO> getJobStatus() {
		return jobstatus;
	}

	public void setJobStatus(List<JobstatusDTO> jobstatus) {
		this.jobstatus = jobstatus;
	}
}
